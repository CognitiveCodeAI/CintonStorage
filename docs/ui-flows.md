## Design System Overview

### Technology Stack

- **Framework**: React 18+ with TypeScript
- **Styling**: Tailwind CSS + shadcn/ui components
- **State Management**: TanStack Query for server state
- **Forms**: React Hook Form + Zod validation
- **Icons**: Lucide React

### Design Principles

1. **Mobile-First**: Yard operations must work on tablets/phones
2. **Keyboard Accessible**: All actions achievable via keyboard
3. **Fast Feedback**: Loading states, optimistic updates
4. **Clear Hierarchy**: Obvious primary actions, scannable layouts
5. **Error Prevention**: Validation before submission, confirmation for destructive actions

### Color Palette

```css
/* Semantic Colors */
--primary: 222 47% 11%;        /* Navy - Primary actions */
--secondary: 210 40% 96%;      /* Light gray - Secondary actions */
--destructive: 0 84% 60%;      /* Red - Destructive actions */

--success: 142 71% 45%;        /* Green - Success states */

/* Status Colors */

--status-hold: 0 84% 60%;          /* Red */
--status-released: 142 71% 45%;    /* Green */
--status-auction: 262 83% 58%;     /* Purple */
```

### Typography


## Component Specifications

### Status Badge Component

```tsx
// components/ui/status-badge.tsx
import { cn } from '@/lib/utils';
import { VehicleCaseStatus } from '@cinton/shared';

const statusConfig: Record<VehicleCaseStatus, { label: string; className: string }> = {
  PENDING_INTAKE: { label: 'Pending', className: 'bg-amber-100 text-amber-800' },
  INTAKE_COMPLETE: { label: 'Intake Done', className: 'bg-blue-100 text-blue-800' },
  STORED: { label: 'Stored', className: 'bg-blue-100 text-blue-800' },
  HOLD: { label: 'Hold', className: 'bg-red-100 text-red-800' },
  RELEASE_ELIGIBLE: { label: 'Ready', className: 'bg-green-100 text-green-800' },
  RELEASED: { label: 'Released', className: 'bg-gray-100 text-gray-800' },
  AUCTION_ELIGIBLE: { label: 'Auction', className: 'bg-purple-100 text-purple-800' },
  AUCTION_LISTED: { label: 'Listed', className: 'bg-purple-100 text-purple-800' },
  SOLD: { label: 'Sold', className: 'bg-green-100 text-green-800' },
  DISPOSED: { label: 'Disposed', className: 'bg-gray-100 text-gray-800' },
};

interface StatusBadgeProps {
  status: VehicleCaseStatus;
  size?: 'sm' | 'md' | 'lg';
}

export function StatusBadge({ status, size = 'md' }: StatusBadgeProps) {
  const config = statusConfig[status];

  return (
    <span
      className={cn(
        'inline-flex items-center rounded-full font-medium',
        config.className,
        {
          'px-2 py-0.5 text-xs': size === 'sm',
          'px-2.5 py-0.5 text-sm': size === 'md',
          'px-3 py-1 text-base': size === 'lg',
        }
      )}
    >
      {config.label}
    </span>
  );
}
```

### Photo Grid Component (Mobile)

```tsx
// components/photo-grid.tsx
import { useState } from 'react';
import { Camera, X, Check } from 'lucide-react';
import { cn } from '@/lib/utils';
import type { DocumentType } from '@cinton/shared';

interface PhotoSlot {
  type: DocumentType;
  label: string;
  required: boolean;
  imageUrl?: string;
}

interface PhotoGridProps {
  slots: PhotoSlot[];
  onCapture: (type: DocumentType) => void;
  onRemove: (type: DocumentType) => void;
}

export function PhotoGrid({ slots, onCapture, onRemove }: PhotoGridProps) {
  const capturedCount = slots.filter(s => s.imageUrl).length;
  const requiredCount = slots.filter(s => s.required).length;

  return (
    <div className="space-y-4">
      <div className="grid grid-cols-3 gap-3">
        {slots.map((slot) => (
          <div
            key={slot.type}
            className={cn(
              'aspect-square rounded-lg border-2 border-dashed',
              'flex flex-col items-center justify-center',
              'cursor-pointer transition-colors',
              slot.imageUrl
                ? 'border-solid border-green-500 bg-green-50'
                : 'border-muted-foreground/25 hover:border-primary'
            )}
            onClick={() => slot.imageUrl ? onRemove(slot.type) : onCapture(slot.type)}
          >
            {slot.imageUrl ? (
              <div className="relative w-full h-full">
                <img
                  src={slot.imageUrl}
                  alt={slot.label}
                  className="w-full h-full object-cover rounded-lg"
                />
                <div className="absolute top-1 right-1 flex gap-1">
                  <button
                    className="bg-green-500 text-white rounded-full p-1"
                    onClick={(e) => { e.stopPropagation(); }}
                  >
                    <Check className="w-3 h-3" />
                  </button>
                  <button
                    className="bg-red-500 text-white rounded-full p-1"
                    onClick={(e) => { e.stopPropagation(); onRemove(slot.type); }}
                  >
                    <X className="w-3 h-3" />
                  </button>
                </div>
              </div>
            ) : (
              <>
                <Camera className="w-6 h-6 text-muted-foreground" />
                <span className="text-xs text-muted-foreground mt-1 text-center px-1">
                  {slot.label}
                </span>
                {slot.required && (
                  <span className="text-xs text-destructive">Required</span>
                )}
              </>
            )}
          </div>
        ))}
      </div>

      <p className="text-sm text-muted-foreground text-center">
        {capturedCount} of {requiredCount} required photos captured
      </p>
    </div>
  );
}
```

## Accessibility Guidelines

### Keyboard Navigation

- All interactive elements must be focusable
- Tab order follows visual layout
- Escape closes modals and dropdowns
- Arrow keys navigate within components
- Enter/Space activate buttons and links

### Screen Reader Support

```tsx
// Example: Accessible status badge
<span
  role="status"
  aria-label={`Vehicle status: ${statusLabel}`}
  className={statusClassName}
>
  {statusLabel}
</span>

// Example: Form with proper labeling
<label htmlFor="vin" className="sr-only">
  Vehicle Identification Number
</label>
<input
  id="vin"
  type="text"
  aria-describedby="vin-help"
  aria-invalid={!!errors.vin}
/>
<span id="vin-help" className="text-sm text-muted-foreground">
  17-character VIN from vehicle plate
</span>
```

### Color Contrast

- All text meets WCAG 2.1 AA (4.5:1 for normal, 3:1 for large)
- Status colors have text alternatives
- Focus indicators visible in all modes
